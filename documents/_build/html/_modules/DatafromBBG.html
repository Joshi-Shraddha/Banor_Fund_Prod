    print('findbyBBGws01')
    print('Ticker: ', Ticker)
    response = requests.get("http://172.16.20.10:8080/InstrumentInfo", params={"Code": Ticker})

    if response.status_code != 200:
        print('BGG is down!')
        return "BGG is down"
    tree = response.text
    mytree = ET.ElementTree(ET.fromstring(tree))
    root = mytree.getroot()

    if not root:
        return "BGG is down"

    values = []
    print("Getting BBG data...........")
